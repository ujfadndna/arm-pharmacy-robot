#### 0.6.5 MaixCam2 实现代码

```python
# ai_cabinet_organizer.py - AI智能药柜整理
import json
import requests
from maix import camera, image, audio, display

# ========== 配置 ==========
LLM_API_URL = "https://api.deepseek.com/v1/chat/completions"  # 或其他LLM API
LLM_API_KEY = "your-api-key"
SYSTEM_PROMPT = """你是智能药柜管理助手..."""  # 见上文

# ========== 药品数据库 ==========
medicine_db = {}  # slot -> medicine_info

def scan_qr_code(img):
    """扫描二维码获取药品信息"""
    qrcodes = img.find_qrcodes()
    if qrcodes:
        data = json.loads(qrcodes[0].payload())
        return data
    return None

def scan_cabinet():
    """扫描整个药柜，更新药品数据库"""
    global medicine_db
    medicine_db = {}

    cam = camera.Camera(320, 240)
    # 移动到各个槽位扫描
    slots = ["A1", "A2", "A3", "B1", "B2", "B3"]

    for slot in slots:
        # 发送移动命令到RA6M5
        send_command(f"MOVE_TO_SLOT:{slot}")
        wait_for_arrival()

        # 拍照扫描
        img = cam.read()
        medicine = scan_qr_code(img)
        if medicine:
            medicine_db[slot] = medicine

    return medicine_db

def call_llm(user_command, cabinet_state):
    """调用云端LLM获取执行计划"""
    prompt = f"""当前药柜状态:
{format_cabinet_state(cabinet_state)}

用户指令: "{user_command}"
"""

    response = requests.post(
        LLM_API_URL,
        headers={"Authorization": f"Bearer {LLM_API_KEY}"},
        json={
            "model": "deepseek-chat",
            "messages": [
                {"role": "system", "content": SYSTEM_PROMPT},
                {"role": "user", "content": prompt}
            ],
            "response_format": {"type": "json_object"}
        },
        timeout=30
    )

    result = response.json()
    plan = json.loads(result["choices"][0]["message"]["content"])
    return plan

def execute_plan(plan):
    """执行LLM生成的计划"""
    for step in plan["steps"]:
        func = step["func"]
        args = step["args"]

        if func == "move_medicine":
            # 发送移动命令到RA6M5
            cmd = f"MOVE:{args['from']}:{args['to']}"
            send_command(cmd)
            wait_for_completion()

        elif func == "speak":
            # 语音播报
            audio.tts(args["text"])

    # 最终播报
    if "summary" in plan:
        audio.tts(plan["summary"])

def main():
    """主循环"""
    while True:
        # 1. 等待语音唤醒
        text = audio.asr_listen(wakeword="小药")
        if not text:
            continue

        print(f"识别到: {text}")

        # 2. 扫描药柜状态
        cabinet_state = scan_cabinet()

        # 3. 调用LLM规划
        plan = call_llm(text, cabinet_state)
        print(f"LLM计划: {json.dumps(plan, ensure_ascii=False)}")

        # 4. 执行计划
        execute_plan(plan)

if __name__ == "__main__":
    main()
```

##### 摇匀功能技术实现

**原理**: 利用现有`trajectory.c`的多点路径规划，生成8字形摇晃轨迹

**代码示例** (在trajectory相关文件中添加):
```c
/**
 * @brief 生成摇匀轨迹
 * @param planner 轨迹规划器
 * @param q_hold 抓药瓶时的关节角度
 * @param amplitude_deg 摇动幅度（度）
 * @param freq_hz 摇动频率（Hz）
 * @param duration_sec 摇动时长（秒）
 */
void generate_shake_trajectory(traj_planner_t *planner,
                                const float q_hold[6],
                                float amplitude_deg,
                                float freq_hz,
                                float duration_sec)
{
    traj_init(planner, 10);  // 10ms周期

    int points = (int)(duration_sec * 100);  // 100Hz采样
    float q_temp[6];

    for (int i = 0; i < points; i++) {
        float t = (float)i / 100.0f;

        memcpy(q_temp, q_hold, sizeof(q_temp));

        // J4和J5联动，画8字形（利萨如图形）
        q_temp[3] = q_hold[3] + amplitude_deg * sinf(2 * M_PI * freq_hz * t);
        q_temp[4] = q_hold[4] + amplitude_deg * sinf(4 * M_PI * freq_hz * t) / 2;

        traj_add_point(planner, q_temp, 10, TRAJ_INTERP_LINEAR);
    }

    // 回到原位
    traj_add_point(planner, q_hold, 200, TRAJ_INTERP_MINIMUM_JERK);
}
```

**调用时机**: LLM返回的JSON中包含`"shake": true`时触发

```json
{
  "action": "fetch",
  "target": "阿莫西林混悬液",
  "cell_id": 5,
  "shake": true,
  "shake_duration": 3.0
}
```

```python
# voice_interaction.py - 全语音交互系统
import edge_tts
import asyncio
from maix import audio, nn

# ========== 配置 ==========
WAKE_WORD = "小药"
TTS_VOICE = "zh-CN-XiaoxiaoNeural"  # 微软小晓，女声

# ========== 语音合成 ==========
async def speak_async(text):
    """异步语音合成并播放"""
    communicate = edge_tts.Communicate(text, TTS_VOICE)
    audio_data = b""
    async for chunk in communicate.stream():
        if chunk["type"] == "audio":
            audio_data += chunk["data"]
    audio.play(audio_data, sample_rate=24000)

def speak(text):
    """同步版本的语音合成"""
    asyncio.run(speak_async(text))

# ========== 语音回应模板 ==========
RESPONSES = {
    "wake": ["我在", "请说", "你好"],
    "confirm_fetch": "好的，正在取{drug}",
    "confirm_organize": "好的，正在整理药柜",
    "complete_fetch": "已取出{drug}，请注意有效期到{expiry}",
    "complete_organize": "整理完成，{summary}",
    "error_not_found": "抱歉，没有找到{drug}",
    "error_asr": "抱歉，没听清，请再说一遍",
    "collision": "检测到障碍，已紧急停止",
    "stop": "已停止",
}

# ========== 主交互类 ==========
class VoiceInteraction:
    def __init__(self, serial):
        self.serial = serial  # UART连接RA6M5
        self.asr = nn.ASR()

    def on_wake(self):
        """唤醒回调"""
        import random
        speak(random.choice(RESPONSES["wake"]))

    def on_command(self, text):
        """命令处理"""
        # 取药命令
        if "取" in text:
            drug = self.extract_drug_name(text)
            if drug:
                speak(RESPONSES["confirm_fetch"].format(drug=drug))
                result = self.execute_fetch(drug)
                if result["success"]:
                    speak(RESPONSES["complete_fetch"].format(
                        drug=drug, expiry=result["expiry"]))
                else:
                    speak(RESPONSES["error_not_found"].format(drug=drug))

        # 整理命令
        elif "整理" in text:
            speak(RESPONSES["confirm_organize"])
            result = self.execute_organize(text)
            speak(RESPONSES["complete_organize"].format(
                summary=result["summary"]))

        # 停止命令
        elif "停" in text:
            self.execute_stop()
            speak(RESPONSES["stop"])

        # 复杂命令 → LLM
        else:
            speak("正在思考...")
            result = self.call_llm(text)
            self.execute_plan(result)
            speak(result.get("summary", "完成"))

    def on_collision(self):
        """碰撞回调 - RA6M5通知"""
        speak(RESPONSES["collision"])

    def execute_fetch(self, drug):
        """执行取药"""
        self.serial.write(f"CMD:FETCH:{drug}\n".encode())
        return self.wait_result()

    def execute_organize(self, instruction):
        """执行整理"""
        plan = self.call_llm(instruction)
        for step in plan["steps"]:
            self.serial.write(f"CMD:{step}\n".encode())
            self.wait_result()
        return plan

    def execute_stop(self):
        """紧急停止"""
        self.serial.write(b"CMD:STOP\n")

    def extract_drug_name(self, text):
        """从文本提取药品名"""
        drugs = ["阿莫西林", "布洛芬", "感冒灵", "板蓝根"]
        for drug in drugs:
            if drug in text:
                return drug
        return None

    def wait_result(self, timeout=30):
        """等待RA6M5执行结果"""
        # 解析UART返回的结果
        pass

    def call_llm(self, text):
        """调用云端LLM"""
        # 见0.6.5节LLM调用代码
        pass

# ========== 主循环 ==========
def main():
    import serial
    ser = serial.Serial("/dev/ttyS1", 115200)
    vi = VoiceInteraction(ser)

    print("语音交互系统启动，等待唤醒词...")

    while True:
        # 等待唤醒词
        if vi.asr.detect_wakeword(WAKE_WORD):
            vi.on_wake()

            # 录音并识别
            text = vi.asr.recognize(timeout=5)
            if text:
                print(f"识别到: {text}")
                vi.on_command(text)
            else:
                speak(RESPONSES["error_asr"])
```

| 0x20 | 任务完成 | result(1) + data(N) | 执行结果反馈 |
| 0x21 | 碰撞通知 | joint_id(1) | 碰撞检测触发 |
| 0x22 | 状态更新 | status(1) + progress(1) | 执行进度 |

**RA6M5 反馈代码**:
```c
// 任务完成时发送反馈
void send_task_complete(uint8_t result, char* drug_name, char* expiry) {
    uint8_t frame[32];
    frame[0] = 0xAA; frame[1] = 0x55;
    frame[2] = 0x20;  // CMD: 任务完成
    // ... 填充数据
    uart_send(frame, len);
}

// 碰撞时发送通知
void send_collision_notify(uint8_t joint_id) {
    uint8_t frame[8] = {0xAA, 0x55, 0x21, 0x01, joint_id, 0x00};
    frame[5] = frame[2] ^ frame[3] ^ frame[4];  // CRC
    uart_send(frame, 6);
}
```

#### 0.9.2 我们的"通用性"

**核心原理**: 同一套硬件 + LLM任务规划 = 不改代码就能做新任务

| 任务 | 用户指令 | 底层动作 |
|------|----------|----------|

> "我们的系统具备**领域通用性**。传统机器人每增加一个任务就需要重新编程，而我们通过**LLM任务规划**实现了**零代码任务扩展**。

**协议设计**（二进制帧，效率高、防粘包）：
```
帧格式:
┌──────┬──────┬─────┬─────┬────────────┬──────┐
│ 0xAA │ 0x55 │ CMD │ LEN │   DATA     │ CRC  │
│ 帧头1 │ 帧头2 │ 命令 │ 长度 │ 数据(变长) │ 校验 │
└──────┴──────┴─────┴─────┴────────────┴──────┘

命令定义:
  CMD=0x01: 视觉位置 (LEN=17)
            DATA: tag_id(1) + x(4) + y(4) + z(4) + theta(4)
  CMD=0x02: 语音命令 (LEN=2)
            DATA: cmd_type(1) + param(1)
            cmd_type: 0x01=FETCH, 0x02=HOME, 0x03=STOP
  CMD=0x10: 状态反馈 (RA6M5→MaixCam2, LEN=1)
            DATA: status(1)
            status: 0x00=READY, 0x01=MOVING, 0x02=DONE, 0xFF=ERROR

CRC计算: XOR校验 (CMD ^ LEN ^ DATA[0] ^ DATA[1] ^ ...)
```

**MaixCam2 发送示例 (Python)**:
```python
import struct

def send_vision_data(serial, tag_id, x, y, z, theta):
    cmd = 0x01
    data = struct.pack('<Bffff', tag_id, x, y, z, theta)  # 小端
    length = len(data)
    crc = cmd ^ length
    for b in data:
        crc ^= b
    frame = bytes([0xAA, 0x55, cmd, length]) + data + bytes([crc])
    serial.write(frame)
```

```c
#include "hal_data.h"

#define SERVO_PERIOD_US     20000   // 20ms (50Hz)
#define SERVO_MIN_US        500     // 0°
#define SERVO_MAX_US        2500    // 180°

#define GRIPPER_OPEN_ANGLE  0       // 张开角度（需实测标定）
#define GRIPPER_CLOSE_ANGLE 90      // 闭合角度（需实测标定）

void servo_init(void) {
    R_GPT_Open(&g_timer_servo_ctrl, &g_timer_servo_cfg);
    R_GPT_Start(&g_timer_servo_ctrl);
}

void servo_set_angle(uint8_t angle) {
    uint32_t pulse_us = SERVO_MIN_US +
                        (uint32_t)(SERVO_MAX_US - SERVO_MIN_US) * angle / 180;
    uint32_t duty = (pulse_us * 100) / SERVO_PERIOD_US;
    R_GPT_DutyCycleSet(&g_timer_servo_ctrl, duty, GPT_IO_PIN_GTIOCA);
}

void gripper_open(void) {
    servo_set_angle(GRIPPER_OPEN_ANGLE);
}

void gripper_close(void) {
    servo_set_angle(GRIPPER_CLOSE_ANGLE);
}
```

```c
    R_IOPORT_PinWrite(&g_ioport_ctrl, RELAY_PUMP_PIN, BSP_IO_LEVEL_LOW);

void vacuum_on(void) {
    // 开泵 + 关阀 = 抽真空
    R_IOPORT_PinWrite(&g_ioport_ctrl, RELAY_PUMP_PIN, BSP_IO_LEVEL_HIGH);
    R_IOPORT_PinWrite(&g_ioport_ctrl, RELAY_VALVE_PIN, BSP_IO_LEVEL_LOW);
}

void vacuum_off(void) {
    // 关泵 + 开阀 = 破真空释放
    R_IOPORT_PinWrite(&g_ioport_ctrl, RELAY_PUMP_PIN, BSP_IO_LEVEL_LOW);
    R_IOPORT_PinWrite(&g_ioport_ctrl, RELAY_VALVE_PIN, BSP_IO_LEVEL_HIGH);
    R_BSP_SoftwareDelay(200, BSP_DELAY_UNITS_MILLISECONDS);
    R_IOPORT_PinWrite(&g_ioport_ctrl, RELAY_VALVE_PIN, BSP_IO_LEVEL_LOW);
```

**PCA 算法实现**:
```python
import numpy as np

def pca_grasp_angle(contour_points):
    """
    PCA主成分分析计算抓取角度

    输入: 轮廓点集 [(x1,y1), (x2,y2), ...]
    输出: 质心(cx,cy), 主轴角度θ (弧度)

    原理:
    1. 计算点集协方差矩阵
    2. 特征值分解找到主成分方向
    3. 最大特征值对应的特征向量即为主轴
    """
    pts = np.array(contour_points, dtype=np.float32)

    # 计算质心
    cx, cy = np.mean(pts[:, 0]), np.mean(pts[:, 1])

    # 中心化
    centered = pts - np.array([cx, cy])

    # 协方差矩阵
    cov = np.cov(centered.T)

    # 特征值分解
    eigenvalues, eigenvectors = np.linalg.eig(cov)

    # 主轴方向 (最大特征值对应的特征向量)
    major_idx = np.argmax(eigenvalues)
    major_axis = eigenvectors[:, major_idx]

    # 计算角度
    theta = np.arctan2(major_axis[1], major_axis[0])

    return (cx, cy), theta


def normalize_angle(theta):
    """角度归一化到 [-π/2, π/2]，避免180°跳变"""
    while theta > np.pi / 2:
        theta -= np.pi
    while theta < -np.pi / 2:
        theta += np.pi
    return theta
```

```python
class SlidingAverageFilter:
    """滑动平均滤波器 - 简单有效"""
    
    def __init__(self, window=5):
        self.buffer = []
        self.window = window
    
    def update(self, value):
        self.buffer.append(value)
        if len(self.buffer) > self.window:
            self.buffer.pop(0)
        return sum(self.buffer) / len(self.buffer)
    
    def reset(self):
        self.buffer = []
```

**卡尔曼滤波实现**（备用/答辩用）:
```python
class KalmanFilter1D:
    """一维卡尔曼滤波器"""

    def __init__(self, q=0.1, r=1.0, initial_value=0):
        """
        参数:
            q: 过程噪声协方差 (越小越相信模型预测)
            r: 测量噪声协方差 (越小越相信测量值)
        """
        self.q = q  # 过程噪声
        self.r = r  # 测量噪声
        self.x = initial_value  # 状态估计
        self.p = 1.0  # 误差协方差

    def update(self, measurement):
        """
        卡尔曼滤波更新步骤

        输入: 当前测量值
        输出: 滤波后的估计值
        """
        # 预测步骤
        # x_pred = x (假设匀速模型)
        # p_pred = p + q
        self.p = self.p + self.q

        # 更新步骤
        # 卡尔曼增益: K = p / (p + r)
        k = self.p / (self.p + self.r)

        # 状态更新: x = x + K * (z - x)
        self.x = self.x + k * (measurement - self.x)

        # 协方差更新: p = (1 - K) * p
        self.p = (1 - k) * self.p

        return self.x

    def reset(self, value=0):
        """重置滤波器状态"""
        self.x = value
        self.p = 1.0


# 创建4个滤波器实例
kf_x = KalmanFilter1D(q=0.05, r=0.5)      # 位置滤波 (相信测量)
kf_y = KalmanFilter1D(q=0.05, r=0.5)
kf_z = KalmanFilter1D(q=0.05, r=0.5)
kf_theta = KalmanFilter1D(q=0.02, r=0.3)  # 角度滤波 (稍微平滑)
```

#### 5.4 碰撞检测代码实现

```c
// collision_detect.c - ZDT_X_V2 碰撞检测模块

#include "zdt_x_v2_driver.h"
#include "modbus_rtu.h"

// 状态标志位定义
#define STATUS_STALLED          0x04    // bit2: 堵转标志
#define STATUS_STALL_PROTECT    0x08    // bit3: 堵转保护已触发

// 碰撞状态
typedef struct {
    bool detected;
    uint8_t joint_id;       // 发生碰撞的关节
    uint16_t current_ma;    // 碰撞时的电流
    uint32_t timestamp;
} collision_info_t;

static collision_info_t g_collision = {0};

/**
 * @brief 碰撞检测轮询 (建议20ms周期调用)
 * @return true=检测到碰撞
 */
bool collision_detect_poll(void) {
    uint8_t status;
    
    // 轮询6个关节
    for (uint8_t id = 1; id <= 6; id++) {
        status = zdt_read_motor_status(id);
        
        // 检查堵转标志 (bit2) 或 堵转保护标志 (bit3)
        if (status & (STATUS_STALLED | STATUS_STALL_PROTECT)) {
            // 记录碰撞信息
            g_collision.detected = true;
            g_collision.joint_id = id;
            g_collision.current_ma = zdt_read_phase_current(id);
            g_collision.timestamp = HAL_GetTick();
            
            // 立即停止所有电机
            zdt_immediate_stop_all();
            
            // 设置系统状态
            set_system_state(STATE_COLLISION_STOP);
            
            // 蜂鸣器报警 (可选)
            buzzer_beep(3, 100);
            
            return true;
        }
    }
    return false;
}

/**
 * @brief 读取电机状态标志位
 */
uint8_t zdt_read_motor_status(uint8_t id) {
    uint8_t tx[8], rx[8];
    
    // 构建Modbus请求帧
    tx[0] = id;
    tx[1] = 0x04;           // 功能码: 读输入寄存器
    tx[2] = 0x00; tx[3] = 0x50;  // 寄存器地址
    tx[4] = 0x00; tx[5] = 0x01;  // 寄存器数量
    modbus_append_crc(tx, 6);
    
    modbus_transceive(tx, 8, rx, 7, 50);
    
    return rx[4];  // 状态字节在响应帧的第5字节
}

/**
 * @brief 广播立即停止
 */
void zdt_immediate_stop_all(void) {
    uint8_t tx[11];
    
    tx[0] = 0x00;           // 广播地址
    tx[1] = 0x10;           // 功能码: 写多个寄存器
    tx[2] = 0x00; tx[3] = 0xFE;  // 寄存器地址
    tx[4] = 0x00; tx[5] = 0x01;  // 寄存器数量
    tx[6] = 0x02;           // 字节数
    tx[7] = 0x98;           // 数据高字节
    tx[8] = 0x00;           // 数据低字节 (多机同步标志=0)
    modbus_append_crc(tx, 9);
    
    modbus_send(tx, 11);    // 广播不等待响应
}

/**
 * @brief 解除堵转保护并恢复
 */
void collision_recover(void) {
    if (!g_collision.detected) return;
    
    // 1. 解除所有关节的堵转保护
    for (uint8_t id = 1; id <= 6; id++) {
        uint8_t tx[8];
        tx[0] = id;
        tx[1] = 0x06;               // 功能码: 写单个寄存器
        tx[2] = 0x00; tx[3] = 0x0E; // 解除堵转保护寄存器
        tx[4] = 0x00; tx[5] = 0x01;
        modbus_append_crc(tx, 6);
        modbus_transceive(tx, 8, NULL, 0, 20);
    }
    
    // 2. 清除碰撞状态
    g_collision.detected = false;
    g_collision.joint_id = 0;
    
    // 3. 恢复系统状态
    set_system_state(STATE_IDLE);
}

/**
 * @brief 获取碰撞信息 (用于显示/记录)
 */
bool collision_get_info(uint8_t *joint, uint16_t *current, uint32_t *time) {
    if (!g_collision.detected) return false;
    
    if (joint)   *joint = g_collision.joint_id;
    if (current) *current = g_collision.current_ma;
    if (time)    *time = g_collision.timestamp;
    return true;
}
```

#### 🥈 方案2: Minimum Jerk 轨迹规划

> **难度**: ⭐⭐ | **演示效果**: ⭐⭐⭐⭐ | **实现时间**: 半天

**原理**: 最小化加加速度(jerk)的积分，生成最平滑的轨迹。五次多项式，起止速度和加速度自动为0。

**对比**:
```
梯形加减速:              Minimum Jerk:
速度 ^                   速度 ^
     |    ___                 |    ___
     |   /   \                |   (   )    ← 全程平滑
     |  /     \               |  /     \
     | /       \              | (       )
     +----------→ 时间        +----------→ 时间
     加速度突变              加加速度最小化
```

**数学公式**:
$$x(t) = x_0 + (x_f - x_0) \cdot (10\tau^3 - 15\tau^4 + 6\tau^5)$$
其中 $\tau = t/T$ 是归一化时间。

**代码实现**:
```c
// minimum_jerk.c - 最小加加速度轨迹规划
#include <math.h>

/**
 * @brief Minimum Jerk 轨迹插值
 * @param t      当前时间 (0 ~ T)
 * @param T      总运动时间
 * @param x0     起始位置
 * @param xf     目标位置
 * @return       当前位置
 */
float minimum_jerk_position(float t, float T, float x0, float xf) {
    if (t <= 0) return x0;
    if (t >= T) return xf;

    float tau = t / T;              // 归一化时间 [0, 1]
    float tau3 = tau * tau * tau;
    float tau4 = tau3 * tau;
    float tau5 = tau4 * tau;

    // 五次多项式: 10τ³ - 15τ⁴ + 6τ⁵
    float s = 10.0f * tau3 - 15.0f * tau4 + 6.0f * tau5;

    return x0 + (xf - x0) * s;
}

/**
 * @brief Minimum Jerk 速度
 */
float minimum_jerk_velocity(float t, float T, float x0, float xf) {
    if (t <= 0 || t >= T) return 0;

    float tau = t / T;
    float tau2 = tau * tau;
    float tau3 = tau2 * tau;
    float tau4 = tau3 * tau;

    // 导数: 30τ² - 60τ³ + 30τ⁴
    float ds = 30.0f * tau2 - 60.0f * tau3 + 30.0f * tau4;

    return (xf - x0) / T * ds;
}

/**
 * @brief 6轴同步 Minimum Jerk 轨迹
 */
typedef struct {
    float start[6];     // 起始关节角
    float end[6];       // 目标关节角
    float T;            // 总时间
    float t;            // 当前时间
} jerk_trajectory_t;

void jerk_traj_init(jerk_trajectory_t* traj, float start[6], float end[6], float duration) {
    for (int i = 0; i < 6; i++) {
        traj->start[i] = start[i];
        traj->end[i] = end[i];
    }
    traj->T = duration;
    traj->t = 0;
}

bool jerk_traj_update(jerk_trajectory_t* traj, float dt, float output[6]) {
    traj->t += dt;

    for (int i = 0; i < 6; i++) {
        output[i] = minimum_jerk_position(traj->t, traj->T,
                                          traj->start[i], traj->end[i]);
    }

    return traj->t >= traj->T;  // 返回是否完成
}
```

**为什么比S曲线更适合这个项目**:
| 对比项 | Minimum Jerk | S曲线 |
|--------|-------------|-------|
| 代码量 | ~30行 | ~150行 |
| 调参 | 只有总时间T | 7个参数 |
| 平滑度 | 数学最优 | 很好 |
| Bug风险 | 低 | 高（7段边界） |

**答辩话术**:
> "轨迹规划采用**最小加加速度（Minimum Jerk）**算法，使用五次多项式生成全局最平滑的轨迹。该算法源自人体运动学研究，能最小化机械冲击，同时代码简洁可靠。"

#### 💡 方案4: 视觉伺服闭环（进阶，时间充裕时考虑）

> **难度**: ⭐⭐⭐⭐ | **演示效果**: ⭐⭐⭐⭐⭐ | **实现时间**: 3-5天

**原理**: 机械臂运动过程中持续接收视觉反馈，实时修正轨迹

```c
// visual_servo.c - 视觉伺服控制
#define VISUAL_SERVO_FREQ   20      // 20Hz
#define POSITION_DEADZONE   5.0f    // 5mm 死区

void visual_servo_task(void *pvParameters) {
    float target_cam[3];    // 视觉目标位置
    float current_cam[3];   // 当前检测位置
    float error[3];
    float Kp = 0.3f;        // 比例增益
    
    while (1) {
        // 等待新的视觉数据
        if (vision_data_available()) {
            vision_get_target(target_cam);
            vision_get_current(current_cam);
            
            // 计算位置误差
            for (int i = 0; i < 3; i++) {
                error[i] = target_cam[i] - current_cam[i];
            }
            
            float error_norm = sqrtf(error[0]*error[0] + error[1]*error[1] + error[2]*error[2]);
            
            // 超过死区才修正
            if (error_norm > POSITION_DEADZONE) {
                float correction[3];
                for (int i = 0; i < 3; i++) {
                    correction[i] = Kp * error[i];
                }
                
                // 应用轨迹修正
                trajectory_apply_correction(correction);
            }
        }
        
        vTaskDelay(pdMS_TO_TICKS(1000 / VISUAL_SERVO_FREQ));
    }
}
```

#### 🔧 方案6: DTC零拷贝CAN接收

**原理**: DTC (Data Transfer Controller) 是 RA6M5 的硬件 DMA，CAN 接收完全不占 CPU

**FSP 配置要点**:
```
1. CAN 模块 → Properties → DTC Transfer Support → Enable
2. 创建 DTC 实例，绑定到 CAN RX 完成事件
3. 配置目标地址为环形缓冲区
```

**代码实现**:
```c
// can_dtc_rx.c - DTC 零拷贝 CAN 接收
#include "r_dtc.h"
#include "r_can.h"

#define CAN_RX_BUFFER_SIZE  64

typedef struct {
    can_frame_t frames[CAN_RX_BUFFER_SIZE];
    volatile uint32_t head;
    volatile uint32_t tail;
} can_rx_ring_t;

static can_rx_ring_t g_can_rx_ring;

// DTC 配置: CAN RX → 环形缓冲区
static transfer_info_t g_dtc_can_rx_info = {
    .transfer_settings_word = TRANSFER_MODE_REPEAT | TRANSFER_SIZE_4_BYTE,
    .p_dest = (void*)g_can_rx_ring.frames,
    .p_src  = (void*)&R_CAN0->MB[0].D,     // CAN 邮箱数据寄存器
    .num_blocks = CAN_RX_BUFFER_SIZE,
    .length = sizeof(can_frame_t) / 4,
};

void can_dtc_init(void) {
    // DTC 初始化
    R_DTC_Open(&g_dtc_ctrl, &g_dtc_cfg);
    R_DTC_Enable(&g_dtc_ctrl);

    // 绑定到 CAN RX 事件
    R_DTC_Reconfigure(&g_dtc_ctrl, &g_dtc_can_rx_info);
}

// 主循环只需要处理缓冲区数据，无需管接收中断
void can_process_rx_buffer(void) {
    while (g_can_rx_ring.head != g_can_rx_ring.tail) {
        can_frame_t* frame = &g_can_rx_ring.frames[g_can_rx_ring.tail];

        // 解析电机状态
        motor_parse_feedback(frame);

        g_can_rx_ring.tail = (g_can_rx_ring.tail + 1) % CAN_RX_BUFFER_SIZE;
    }
}
```

#### 🔧 方案9: 柔顺放置（基于堵转保护）

> **难度**: ⭐ | **演示效果**: ⭐⭐⭐ | **实现时间**: 2小时

**原理**: 利用ZDT_X_V2驱动器的堵转保护功能实现柔顺放置。配置较低的堵转阈值，接触即停。

**为什么不用软件读电流**:
```
软件读电流方案的问题：
1. Modbus-RTU读电流需要20-50ms
2. 下降速度5mm/s，50ms就是0.25mm
3. 检测到接触时，已经压下去了

ZDT堵转保护的优势：
1. 驱动器硬件检测，响应<10ms
2. 检测到即停止，无延迟
3. 不需要主控参与
```

**实现方案**：临时调低堵转保护阈值

**代码实现**:
```c
// compliant_place.c - 基于堵转保护的柔顺放置
#include "zdt_x_v2_driver.h"

// 正常运动时的堵转参数
#define STALL_CURRENT_NORMAL    600     // mA
#define STALL_TIME_NORMAL       80      // ms

// 柔顺放置时的堵转参数（更灵敏）
#define STALL_CURRENT_SOFT      300     // mA - 降低阈值
#define STALL_TIME_SOFT         30      // ms - 缩短检测时间

/**
 * @brief 配置堵转保护参数
 */
void zdt_set_stall_params(uint8_t id, uint16_t current_ma, uint16_t time_ms) {
    // 寄存器 0x0080: 堵转电流阈值
    // 寄存器 0x0081: 堵转时间阈值
    zdt_write_register(id, 0x0080, current_ma);
    zdt_write_register(id, 0x0081, time_ms);
}

/**
 * @brief 柔顺放置流程
 */
bool compliant_place(float place_x, float place_y, float place_z_min) {
    uint8_t z_joint_id = 3;  // Z轴关节ID（根据实际调整）

    // 1. 移动到放置位置上方
    robot_move_to_xyz(place_x, place_y, place_z_min + 50.0f);
    wait_motion_complete();

    // 2. 临时调低Z轴堵转阈值（更灵敏）
    zdt_set_stall_params(z_joint_id, STALL_CURRENT_SOFT, STALL_TIME_SOFT);

    // 3. 缓慢下降到目标位置
    robot_move_z_to(place_z_min, 2.0f);  // 2mm/s 慢速下降

    // 4. 等待运动完成或堵转触发
    uint32_t timeout = 10000;  // 10秒超时
    uint32_t start = HAL_GetTick();
    bool contact = false;

    while (HAL_GetTick() - start < timeout) {
        uint8_t status = zdt_read_motor_status(z_joint_id);

        if (status & STATUS_STALLED) {
            // 堵转触发 = 接触到桌面
            contact = true;
            break;
        }

        if (status & STATUS_ARRIVED) {
            // 到达目标位置，未接触
            break;
        }

        vTaskDelay(pdMS_TO_TICKS(10));
    }

    // 5. 恢复正常堵转参数
    zdt_set_stall_params(z_joint_id, STALL_CURRENT_NORMAL, STALL_TIME_NORMAL);

    // 6. 解除堵转保护（如果触发了）
    if (contact) {
        zdt_release_stall_protect(z_joint_id);

        // 稍微抬起释放压力
        robot_move_z_relative(2.0f, 5.0f);
        wait_motion_complete();

        // 打开夹爪
        gripper_open();
        vTaskDelay(pdMS_TO_TICKS(300));

        // 抬起
        robot_move_z_relative(30.0f, 20.0f);
    }

    return contact;
}
```

**演示效果**:
```
1. 机械臂携带药盒移动到药篮上方
2. 缓慢下降（2mm/s）...
3. 药盒接触药篮底部 → 驱动器堵转保护触发 → 电机立即停止
4. 夹爪打开，药盒轻轻落下
5. 机械臂抬起

评委看到的：药盒是"放"下去的，不是"砸"下去的
```

**答辩话术**:
> "放置药盒时，我们利用ZDT驱动器的堵转保护功能实现柔顺放置。临时调低堵转电流阈值，使驱动器对接触更敏感。当药盒接触桌面时，驱动器硬件检测到阻力，10ms内自动停止电机。这比软件轮询电流更快更可靠。"

#### 🔧 方案11: 状态机帧解析器（防粘包）

> **难度**: ⭐⭐ | **稳定性提升**: ⭐⭐⭐⭐⭐ | **实现时间**: 半天

**问题**: MaixCam2高频发送数据时，UART可能收到粘包（半帧+一帧半），直接解析会导致数据错乱，机械臂可能疯转。

**解决方案**: 环形缓冲区 + 状态机逐字节解析

**帧格式设计**:
```
┌──────┬──────┬─────┬─────┬────────────┬──────┐
│ 0xAA │ 0x55 │ CMD │ LEN │   DATA     │ CRC  │
│ 帧头1 │ 帧头2 │ 命令 │ 长度 │ 数据(变长) │ 校验 │
└──────┴──────┴─────┴─────┴────────────┴──────┘
```

**代码实现**:
```c
// frame_parser.c - 状态机帧解析器
#include <stdint.h>
#include <stdbool.h>
#include <string.h>

#define RING_BUF_SIZE   256
#define FRAME_MAX_DATA  32

// 环形缓冲区
typedef struct {
    uint8_t buf[RING_BUF_SIZE];
    volatile uint16_t head;
    volatile uint16_t tail;
} ring_buffer_t;

static ring_buffer_t g_rx_ring = {0};

// 解析状态机
typedef enum {
    STATE_WAIT_HEAD1,
    STATE_WAIT_HEAD2,
    STATE_WAIT_CMD,
    STATE_WAIT_LEN,
    STATE_WAIT_DATA,
    STATE_WAIT_CRC
} parse_state_t;

// 解析后的帧
typedef struct {
    uint8_t cmd;
    uint8_t len;
    uint8_t data[FRAME_MAX_DATA];
} parsed_frame_t;

static parse_state_t g_state = STATE_WAIT_HEAD1;
static uint8_t g_cmd, g_len, g_data_idx;
static uint8_t g_frame_data[FRAME_MAX_DATA];

/**
 * @brief DTC/中断回调 - 把数据塞进环形缓冲区
 */
void uart_rx_callback(uint8_t* data, uint16_t size) {
    for (uint16_t i = 0; i < size; i++) {
        g_rx_ring.buf[g_rx_ring.head] = data[i];
        g_rx_ring.head = (g_rx_ring.head + 1) % RING_BUF_SIZE;

        // 溢出检测（可选：丢弃旧数据）
        if (g_rx_ring.head == g_rx_ring.tail) {
            g_rx_ring.tail = (g_rx_ring.tail + 1) % RING_BUF_SIZE;
        }
    }
}

/**
 * @brief CRC8校验
 */
static uint8_t calc_crc8(uint8_t cmd, uint8_t len, uint8_t* data) {
    uint8_t crc = cmd ^ len;
    for (uint8_t i = 0; i < len; i++) {
        crc ^= data[i];
    }
    return crc;
}

/**
 * @brief 状态机解析 - 主循环调用
 * @param out  解析成功时输出帧数据
 * @return true=解析到完整帧
 */
bool frame_parser_poll(parsed_frame_t* out) {
    while (g_rx_ring.tail != g_rx_ring.head) {
        uint8_t byte = g_rx_ring.buf[g_rx_ring.tail];
        g_rx_ring.tail = (g_rx_ring.tail + 1) % RING_BUF_SIZE;

        switch (g_state) {
            case STATE_WAIT_HEAD1:
                if (byte == 0xAA) g_state = STATE_WAIT_HEAD2;
                break;

            case STATE_WAIT_HEAD2:
                g_state = (byte == 0x55) ? STATE_WAIT_CMD : STATE_WAIT_HEAD1;
                break;

            case STATE_WAIT_CMD:
                g_cmd = byte;
                g_state = STATE_WAIT_LEN;
                break;

            case STATE_WAIT_LEN:
                g_len = byte;
                if (g_len > FRAME_MAX_DATA) {
                    g_state = STATE_WAIT_HEAD1;  // 长度异常，重新找帧头
                } else {
                    g_data_idx = 0;
                    g_state = (g_len > 0) ? STATE_WAIT_DATA : STATE_WAIT_CRC;
                }
                break;

            case STATE_WAIT_DATA:
                g_frame_data[g_data_idx++] = byte;
                if (g_data_idx >= g_len) {
                    g_state = STATE_WAIT_CRC;
                }
                break;

            case STATE_WAIT_CRC:
                if (byte == calc_crc8(g_cmd, g_len, g_frame_data)) {
                    // CRC正确，输出帧
                    out->cmd = g_cmd;
                    out->len = g_len;
                    memcpy(out->data, g_frame_data, g_len);
                    g_state = STATE_WAIT_HEAD1;
                    return true;
                }
                // CRC错误，丢弃，重新找帧头
                g_state = STATE_WAIT_HEAD1;
                break;
        }
    }
    return false;
}

/**
 * @brief 主循环中处理帧
 */
void process_maixcam_data(void) {
    parsed_frame_t frame;

    while (frame_parser_poll(&frame)) {
        switch (frame.cmd) {
            case CMD_VISION_POS:  // 视觉位置数据
                // 解析 x, y, z, theta
                float* pos = (float*)frame.data;
                vision_update_target(pos[0], pos[1], pos[2], pos[3]);
                break;

            case CMD_VOICE_CMD:   // 语音命令
                handle_voice_command(frame.data[0]);
                break;

            default:
                break;
        }
    }
}
```

**为什么必须用状态机**:
```
场景：MaixCam2 以 20Hz 发送位置数据

错误做法（直接解析）:
  收到: [AA 55 01 08 ...数据... CRC] [AA 55 01 08 ...
        ^^^^^^^^^^^^^^^^^^^^^^^^^^ ^^^^^^^^^^^^^^^^
             完整帧                    半帧（下次才收到剩下的）

  直接解析 → 第二帧数据错位 → 机械臂疯转

正确做法（状态机）:
  收到半帧 → 状态停在 STATE_WAIT_DATA
  收到剩余 → 继续解析 → 完整帧输出
  无论怎么粘包都能正确分帧
```

**答辩话术**:
> "MaixCam2与RA6M5的UART通信采用**状态机帧解析器**。数据先存入环形缓冲区，再由状态机逐字节解析。这种设计能正确处理粘包和半帧情况，确保即使在高频通信下也不会出现数据错乱。"
